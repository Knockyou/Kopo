#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main()
{
    int m = 0, n = 0;

    if (!scanf("%d", &m))
    {
        printf("정수의 값을 입력 해주세요");
        return 0;
    }

    if (!scanf("%d", &n))
    {
        printf("정수의 값을 입력 해주세요");
        return 0;
    }

    if (n < m)
    {
        printf("두번째 정수는 첫번째 정수보다 큰수이어야 합니다");
        return 0;
    }

    char temp[500] = { 0, };

    for (int i = m; i <= n; i++)
    {
        sprintf(temp, "%d", i);

        if (i % 7 == 0 || strchr(temp, '7') != NULL)
            continue;

        if (i % 5 == 0)
        {
            printf("%d ", i);
            break;
        }

        printf("%d ", i);
    }

    printf("\n");
    return 0;
}
