#include <stdio.h>
#include <stdlib.h>
#include <time.h>
                    
int main()
{   
    // 2과제
    int num = 0;
    if(!scanf("%d", &num))
    {
        printf("정수의 값을 입력하세요");
        return 0;
    }
    
    printf("확인용 : %d\n", num);
    
    if(num < 0)
    {
        printf("양수의 값을 입력하세요"); 
        return 0;
    }
    else if(num / 1000 >= 10 || sizeof(num) != sizeof(int))
    {
        printf("3 자리의 정수만 입력 가능 합니다"); 
        return 0;
    }
    
    int hFix = num / 100;
    int tFix = (num - (hFix * 100)) / 10;
    int sFix = num - ((hFix * 100) + (tFix * 10));
    
    printf("자리 수의 합 : %d\n", hFix + tFix + sFix);
                   
    return 0;
}
